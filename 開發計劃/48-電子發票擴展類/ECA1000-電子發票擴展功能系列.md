## 三、後端 API 設計

### 3.1 API 端點列表

#### 3.1.1 電子發票擴展處理
- **HTTP 方法**: `POST`
- **路徑**: `/api/v1/einvoices/extensions/process`
- **說明**: 處理電子發票擴展功能
- **請求格式**:
  ```json

    "invoiceId": 1,
    "extensionType": "ADVANCED_PROCESSING",
    "data": {}
  }
  ```
- **回應格式**:

    "success": true,
    "code": 200,
    "message": "處理成功",
    "data": {

      "status": "SUCCESS"

### 3.2 後端實作類別

#### 3.2.1 Controller: `EInvoiceExtensionController.cs`
```csharp
namespace RSL.IMS3.Api.Controllers
{
    [ApiController]
    [Route("api/v1/einvoices/extensions")]
    [Authorize]
    public class EInvoiceExtensionController : ControllerBase
    {
        private readonly IEInvoiceExtensionService _extensionService;
        
        public EInvoiceExtensionController(IEInvoiceExtensionService extensionService)
        {
            _extensionService = extensionService;
        }
        
        [HttpPost("process")]
        public async Task<ActionResult<ApiResponse<EInvoiceExtensionDto>>> ProcessExtension([FromBody] ProcessExtensionDto dto)
        {
            var result = await _extensionService.ProcessExtensionAsync(dto);
            return Ok(new ApiResponse<EInvoiceExtensionDto> { Success = true, Data = result });
        }
        
        [HttpGet]
        public async Task<ActionResult<ApiResponse<PagedResult<EInvoiceExtensionDto>>>> GetExtensions([FromQuery] EInvoiceExtensionQueryDto query)
        {
            var result = await _extensionService.GetExtensionsAsync(query);
            return Ok(new ApiResponse<PagedResult<EInvoiceExtensionDto>> { Success = true, Data = result });
        }
    }
}
```

### 4.3 API 呼叫 (`einvoiceExtension.api.ts`)
```typescript
import request from '@/utils/request';

export interface ProcessExtensionDto {
  invoiceId: number;
  extensionType: string;
  data?: Record<string, any>;
}

export interface EInvoiceExtensionQueryDto {
  pageIndex: number;
  pageSize: number;
  invoiceId?: number;
  extensionType?: string;
}

export interface EInvoiceExtensionDto {
  extensionId: number;
  invoiceId: number;
  extensionType: string;
  extensionData?: Record<string, any>;
  createdAt: string;
}

// 處理電子發票擴展功能
export const processExtension = (dto: ProcessExtensionDto) => {
  return request.post<ApiResponse<EInvoiceExtensionDto>>('/api/v1/einvoices/extensions/process', dto);
};

// 查詢電子發票擴展資料
export const getExtensions = (query: EInvoiceExtensionQueryDto) => {
  return request.get<ApiResponse<PagedResult<EInvoiceExtensionDto>>>('/api/v1/einvoices/extensions', { params: query });
};
```
